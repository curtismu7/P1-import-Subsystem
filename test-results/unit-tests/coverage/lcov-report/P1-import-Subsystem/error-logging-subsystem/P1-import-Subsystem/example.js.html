/**
 * Error Logging Subsystem Example
 * 
 * Demonstrates the Winston-based Error Logging Subsystem with:
 * - Clear log entry delineation
 * - Colorized console output
 * - Structured file logging
 * - Error handling and reporting
 * - Different log levels and contexts
 */
 
import { createErrorLoggingSystem, ErrorFactory } from './index.js';
 
/**
 * Example usage of the Error Logging Subsystem
 */
async function demonstrateErrorLoggingSystem() {
    console.log('\n🚀 Starting Error Logging Subsystem Demonstration\n');
    
    // Create the error logging system with Winston
    const { errorManager, logManager } = createErrorLoggingSystem({
        logLevel: 'debug',
        enableConsole: true,
        enableFile: true,
        logDir: 'logs',
        logFileName: 'demo.log',
        maxFileSize: 5 * 1024 * 1024, // 5MB
        maxFiles: 3,
        enableColors: true,
        enableErrorReporting: false // Disable UI reporting for this demo
    });
    
    // Get a logger for this demo
    const logger = logManager.getLogger('DemoApp', {
        version: '1.0.0',
        environment: 'development'
    });
    
    console.log('📝 Demonstrating different log levels with clear delineation:\n');
    
    // Demonstrate different log levels
    logger.debug('This is a debug message', {
        operation: 'demonstration',
        step: 1,
        details: {
            message: 'Debug messages help with troubleshooting',
            visibility: 'development only'
        }
    });
    
    logger.info('Application started successfully', {
        operation: 'startup',
        step: 2,
        config: {
            port: 4000,
            environment: 'development',
            features: ['auth', 'import', 'export']
        }
    });
    
    logger.warn('Configuration value missing, using default', {
        operation: 'configuration',
        step: 3,
        missing: 'API_TIMEOUT',
        default: 30000,
        impact: 'low'
    });
    
    // Demonstrate error logging with stack traces
    try {
        throw new Error('Simulated database connection error');
    } catch (error) {
        logger.error('Database connection failed', error, {
            operation: 'database',
            step: 4,
            database: 'postgresql',
            host: 'localhost',
            port: 5432,
            retryAttempt: 1
        });
    }
    
    // Demonstrate fatal error logging
    logger.fatal('Critical system failure detected', new Error('Out of memory'), {
        operation: 'system',
        step: 5,
        memoryUsage: '95%',
        availableMemory: '128MB',
        requiredMemory: '2GB',
        action: 'immediate_shutdown_required'
    });
    
    console.log('\n🔥 Demonstrating error handling with different error types:\n');
    
    // Demonstrate different error types
    const validationError = errorManager.createValidationError(
        'Email address is required',
        'email',
        '',
        { userId: 123, form: 'registration' }
    );
    errorManager.handleError(validationError);
    
    const networkError = errorManager.createNetworkError(
        'Failed to connect to API server',
        'https://api.example.com/users',
        'GET',
        503,
        { timeout: 30000, retries: 3 }
    );
    errorManager.handleError(networkError);
    
    const fileProcessingError = errorManager.createFileProcessingError(
        'Invalid CSV format detected',
        'users.csv',
        1024000,
        { lineNumber: 42, expectedColumns: 5, actualColumns: 3 }
    );
    errorManager.handleError(fileProcessingError);
    
    // Demonstrate child logger with additional context
    const importLogger = logManager.getLogger('ImportService', {
        module: 'user-import',
        sessionId: 'sess_123456789'
    });
    
    importLogger.info('Starting user import process', {
        fileName: 'users.csv',
        totalRecords: 1500,
        batchSize: 100
    });
    
    importLogger.warn('Duplicate email detected, skipping record', {
        recordNumber: 245,
        email: 'john.doe@example.com',
        action: 'skip'
    });
    
    importLogger.info('User import completed successfully', {
        totalRecords: 1500,
        imported: 1485,
        skipped: 15,
        duration: '2.3 seconds'
    });
    
    console.log('\n📊 Demonstrating structured logging with complex data:\n');
    
    // Demonstrate complex structured logging
    const apiLogger = logManager.getLogger('APIService');
    
    apiLogger.info('API request processed', {
        request: {
            method: 'POST',
            url: '/api/v1/users',
            headers: {
                'content-type': 'application/json',
                'authorization': 'Bearer [REDACTED]',
                'user-agent': 'PingOne-Import-Tool/1.0.0'
            },
            body: {
                email: 'user@example.com',
                firstName: 'John',
                lastName: 'Doe'
            }
        },
        response: {
            status: 201,
            headers: {
                'content-type': 'application/json',
                'x-request-id': 'req_abc123'
            },
            body: {
                id: 'user_xyz789',
                email: 'user@example.com',
                createdAt: '2024-01-15T10:30:00Z'
            }
        },
        performance: {
            duration: 245,
            dbQueries: 2,
            cacheHits: 1,
            cacheMisses: 0
        }
    });
    
    console.log('\n⚡ Demonstrating error context enrichment:\n');
    
    // Demonstrate error with rich context
    try {
        // Simulate a complex operation failure
        throw new Error('Population creation failed due to quota exceeded');
    } catch (error) {
        errorManager.handleError(error, {
            component: 'PopulationService',
            operation: 'createPopulation',
            user: {
                id: 'user_123',
                email: 'admin@example.com',
                role: 'administrator'
            },
            request: {
                populationName: 'Test Population',
                description: 'A test population for development',
                maxUsers: 1000
            },
            system: {
                currentPopulations: 5,
                maxPopulations: 5,
                quotaType: 'free_tier'
            },
            timestamp: new Date().toISOString(),
            correlationId: 'corr_abc123def456'
        });
    }
    
    console.log('\n✅ Demonstration completed! Check the logs/ directory for file output.\n');
    
    // Flush all logs before exiting
    logManager.flush();
    
    // Wait a moment for async logging to complete
    await new Promise(resolve => setTimeout(resolve, 1000));
    
    console.log('📁 Log files created:');
    console.log('  - logs/demo.log (all logs)');
    console.log('  - logs/error.log (errors only)');
    console.log('  - logs/exceptions.log (uncaught exceptions)');
    console.log('  - logs/rejections.log (unhandled promise rejections)');
    
    return { errorManager, logManager };
}
 
/**
 * Demonstrate error history and management
 */
function demonstrateErrorHistory(errorManager) {
    console.log('\n📚 Error History Management:\n');
    
    // Get error history
    const errorHistory = errorManager.getErrorHistory(5);
    console.log(`Found ${errorHistory.length} errors in history:`);
    
    errorHistory.forEach((error, index) => {
        console.log(`${index + 1}. [${error.severity.toUpperCase()}] ${error.code}: ${error.message}`);
        console.log(`   Category: ${error.category}, Recovery: ${error.recoveryStrategy}`);
        console.log(`   Time: ${error.timestamp}`);
        if (Object.keys(error.context).length > 0) {
            console.log(`   Context: ${JSON.stringify(error.context, null, 2)}`);
        }
        console.log('');
    });
}
 
// Run the demonstration
if (import.meta.url === `file://${process.argv[1]}`) {
    demonstrateErrorLoggingSystem()
        .then(({ errorManager }) => {
            demonstrateErrorHistory(errorManager);
            console.log('🎉 All demonstrations completed successfully!');
        })
        .catch((error) => {
            console.error('❌ Demonstration failed:', error);
            process.exit(1);
        });
}
